禁止 toFixed 四舍五入​

Object.defineProperty(Number.prototype, 'toFixed', {
  value(fractionDigits = 0) {
    const number = this.valueOf()
    const factor = Math.pow(10, fractionDigits)
    return String(Math.trunc(number * factor) / factor)
  },
  enumerable: false,
  writable: true,
  configurable: true
})
